icon_reset                = Reset
icon_delete               = Delete

confirm_reset_all         = the whole counter
success                   = Success
error                     = Error
msg_success_reset_all     = Counter was successfully resetted.
msg_error_reset_all       = An error occured while resetting counter.
msg_success_reset_element = {PARAM1} was successfully resetted.
msg_error_reset_element   = An error occured while resetting {PARAM1}.
data_label                = Label

data_action               = Action
